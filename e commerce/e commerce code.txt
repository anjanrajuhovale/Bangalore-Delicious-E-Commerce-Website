  let cart = [];

  const items = {

    ],

      { name: 'STRAWBERRY', price: 120, img: 'strawberry.jpg' },
      { name: 'BANANA', price: 60, img: 'banana.jpg' },
      { name: 'ORANGE', price: 90, img: 'orange.jpg' },
      { name: 'POMEGRANATES', price: 110, img: 'pomegranates.jpg' }
    ],
    vegetables: [
      { name: 'BRINJAL', price: 40, img: 'brinjial.jpg' },
      { name: 'CARROT', price: 60, img: 'carrot.jpg' },
      { name: 'POTATO', price: 30, img: 'potato.jpg' },
      { name: 'TOMATO', price: 35, img: 'tomato.jpg' }
    ],
    dairy: [
      { name: 'MILK', price: 50, img: 'milk.jpg' },
      { name: 'ICE CREAM', price: 120, img: 'icecream.jpg' },
      { name: 'CHEESE', price: 90, img: 'cheese.jpg' },
      { name: 'BUTTER', price: 70, img: 'butter.jpg' }
    ]
  };

  function showHome() {
    document.getElementById("home-info").style.display = "block";
    document.getElementById("categories").innerHTML = "";
    document.getElementById("cart-section").style.display = "none";
  }

  function showLogin() {
    document.getElementById("login-modal").style.display = "flex";
  }

  function showRegister() {
    document.getElementById("register-modal").style.display = "flex";
  }

  function closeModal() {
    document.querySelectorAll(".modal").forEach(m => m.style.display = "none");
  }

  document.getElementById("register-form").addEventListener("submit", e => {
    e.preventDefault();
    const user = {
      name: document.getElementById("register-name").value,
      email: document.getElementById("register-email").value,
      password: document.getElementById("register-password").value
    };
    localStorage.setItem("user", JSON.stringify(user));
    alert("Registered! Please login.");
    closeModal();
  });

  document.getElementById("login-form").addEventListener("submit", e => {
    e.preventDefault();
    const email = document.getElementById("login-email").value;
    const password = document.getElementById("login-password").value;
    const user = JSON.parse(localStorage.getItem("user"));
    if (user && user.email === email && user.password === password) {
      document.getElementById("main-content").style.display = "block";
      document.getElementById("category-nav").style.display = "block";
      document.getElementById("nav-menu").style.display = "none";
      document.getElementById("home-info").style.display = "block";
      closeModal();
    } else {
      alert("Invalid credentials");
    }
  });

  function signOut() {
    document.getElementById("main-content").style.display = "none";
    document.getElementById("nav-menu").style.display = "flex";
    document.getElementById("category-nav").style.display = "none";
    alert("Signed out.");
  }

  function showCategory(category) {
    const area = document.getElementById("categories");
    area.innerHTML = "";
    document.getElementById("home-info").style.display = "none";
    document.getElementById("cart-section").style.display = "none";
    items[category].forEach(item => {
      area.innerHTML += `
        <div class="item-card">
          <img src="${item.img}" alt="${item.name}">
          <h3>${item.name}</h3>
          <p>₹${item.price}</p>
          <button onclick="addToCart('${item.name}', ${item.price})">Add to Cart</button>
        </div>`;
    });
  }

  function showCart() {
    document.getElementById("categories").innerHTML = "";
    document.getElementById("home-info").style.display = "none";
    document.getElementById("cart-section").style.display = "block";
    renderCart();
  }

  function addToCart(name, price) {
    cart.push({ name, price });
    renderCart();
  }

  function renderCart() {
    const list = document.getElementById("cart-items");
    list.innerHTML = "";
    let total = 0;
    cart.forEach(item => {
      const li = document.createElement("li");
      li.textContent = `${item.name} - ₹${item.price}`;
      list.appendChild(li);
      total += item.price;
    });
    document.getElementById("cart-total").textContent = total.toFixed(2);
  }

  function checkout() {
    if (cart.length === 0) alert("Your cart is empty.");
    else {
      alert("Order placed successfully!");
      cart = [];
      renderCart();
    }
  }
